```python
import numpy as np
import matplotlib.pyplot as plt
import random

def generar_caminata_aleatoria(pasos):
    """
    Genera las coordenadas (x, y) para una caminata aleatoria en 2D.

    La caminata empieza en (0, 0). En cada paso, el movimiento puede ser:
    (1, 0), (-1, 0), (0, 1), o (0, -1).

    :param pasos: El n칰mero de pasos en la caminata.
    :return: Dos arrays de NumPy: coords_x y coords_y.
    """
    # Inicializar arrays para almacenar los cambios de posici칩n (delta_x y delta_y)
    # # La dimensi칩n es igual al n칰mero de pasos.
    delta_x = np.zeros(pasos)
    delta_y = np.zeros(pasos)

    # # Las posibles direcciones de movimiento son 4: Norte, Sur, Este, Oeste.
    direcciones = [(1, 0), (-1, 0), (0, 1), (0, -1)]

    # # Generar los pasos aleatorios
    for i in range(pasos):
        # # Seleccionar una direcci칩n aleatoria de la lista de direcciones
        dx, dy = random.choice(direcciones)
        delta_x[i] = dx
        delta_y[i] = dy

    # # Calcular la posici칩n acumulada sumando los deltas. 
    # # # `np.cumsum` calcula la suma acumulativa de los elementos del array.
    # # # Se a침ade un punto inicial (0, 0) al principio de la caminata.
    coords_x = np.concatenate([[0], np.cumsum(delta_x)])
    coords_y = np.concatenate([[0], np.cumsum(delta_y)])
    
    # # Retorna las coordenadas de la caminata, incluyendo el punto de partida (0, 0).
    return coords_x, coords_y

def visualizar_caminata_aleatoria(coords_x, coords_y):
    """
    Crea un gr치fico que visualiza la caminata aleatoria generada.

    :param coords_x: Array de coordenadas x.
    :param coords_y: Array de coordenadas y.
    """
    # 1. Configuraci칩n del gr치fico
    plt.figure(figsize=(10, 8)) # Define el tama침o de la figura (ancho, alto)
    
    # 2. Dibujar la caminata
    # # `plt.plot` dibuja las l칤neas conectando todos los puntos (x, y).
    plt.plot(coords_x, coords_y, 
             marker='.',       # Muestra un punto en cada posici칩n
             markersize=3,     # Tama침o peque침o para el punto
             linestyle='-',    # L칤nea continua
             linewidth=0.5,    # Grosor de la l칤nea
             alpha=0.7)        # Transparencia

    # 3. Marcar los puntos clave
    # # Punto de inicio (Start)
    plt.plot(coords_x[0], coords_y[0], 'go', # 'go' = green circle
             markersize=8, 
             label='Inicio (0, 0)')
             
    # # Punto final (End)
    plt.plt(coords_x[-1], coords_y[-1], 'ro', # 'ro' = red circle
             markersize=8, 
             label=f'Fin ({coords_x[-1]}, {coords_y[-1]})')

    # 4. Etiquetas y T칤tulos
    plt.title(f"Simulaci칩n de Caminata Aleatoria en 2D ({len(coords_x) - 1} Pasos)", fontsize=16)
    plt.xlabel("Posici칩n X", fontsize=12)
    plt.ylabel("Posici칩n Y", fontsize=12)
    
    # 5. Configuraci칩n de la cuadr칤cula y leyenda
    plt.grid(True, linestyle='--', alpha=0.6) # A침ade una cuadr칤cula
    plt.legend(loc='best') # Muestra la leyenda en la mejor posici칩n
    plt.gca().set_aspect('equal', adjustable='box') # Fuerza que los ejes tengan la misma escala

    # 6. Mostrar el gr치fico
    plt.show()

# --- Bloque Principal de Ejecuci칩n ---

# # Define el n칰mero de pasos para la simulaci칩n (cambia este valor para mayor complejidad)
NUMERO_DE_PASOS = 5000

print("# --- SIMULADOR DE CAMINATA ALEATORIA ---")
print(f"# Generando una caminata de {NUMERO_DE_PASOS} pasos...")

# # 1. Generar los datos
x, y = generar_caminata_aleatoria(NUMERO_DE_PASOS)

# # 2. Visualizar los datos
print("# Visualizando el camino...")
visualizar_caminata_aleatoria(x, y)

print("# Simulaci칩n completada.")
print("# ---------------------------------------")
```
