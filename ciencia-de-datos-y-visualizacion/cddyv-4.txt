# El programa lee un archivo, procesa el texto para limpiarlo y luego cuenta la ocurrencia de cada palabra.

# Utiliza el m칩dulo `collections` para la tarea de contar de forma eficiente.

```python
import re
from collections import Counter
import string

def contador_de_palabras(nombre_archivo):
    """
    Cuenta la frecuencia de cada palabra en un archivo de texto.

    :param nombre_archivo: El nombre (ruta) del archivo de texto a procesar.
    :return: Un objeto Counter con las palabras y sus frecuencias, o None si hay un error.
    """
    # Manejo de errores en caso de que el archivo no exista o no se pueda leer
    try:
        # Abre el archivo en modo lectura ('r') y codificaci칩n UTF-8
        with open(nombre_archivo, 'r', encoding='utf-8') as archivo:
            # Lee todo el contenido del archivo
            texto = archivo.read()
    except FileNotFoundError:
        # Mensaje si el archivo no se encuentra
        print(f"Error: El archivo '{nombre_archivo}' no fue encontrado.")
        return None
    except Exception as e:
        # Mensaje para otros errores de lectura
        print(f"Error al leer el archivo: {e}")
        return None

    # --- Preprocesamiento del texto ---
    
    # 1. Convertir todo el texto a min칰sculas para tratar 'Palabra' y 'palabra' como la misma
    # # Esto es crucial para un conteo de frecuencia preciso.
    texto_minusculas = texto.lower()

    # 2. Eliminar signos de puntuaci칩n. Se reemplazan por un espacio.
    # # Se construye una tabla de traducci칩n para reemplazar la puntuaci칩n con None (eliminaci칩n efectiva).
    # # Se excluyen los guiones (hiphen) si deseas mantener palabras compuestas, pero
    # # # para un conteo simple, es mejor eliminarlos o tratarlos. Aqu칤 se eliminan.
    traductor = str.maketrans('', '', string.punctuation)
    texto_sin_puntuacion = texto_minusculas.translate(traductor)
    
    # # Alternativa para una limpieza m치s estricta (usando regex):
    # # texto_limpio = re.sub(r'[^\w\s]', '', texto_minusculas) # Esto mantiene guiones bajos

    # 3. Dividir el texto en una lista de palabras.
    # # La funci칩n `split()` sin argumentos divide por cualquier espacio en blanco (espacios, saltos de l칤nea, tabuladores).
    palabras = texto_sin_puntuacion.split()

    # 4. Contar la frecuencia de las palabras
    # # `collections.Counter` es la forma m치s eficiente y legible de hacer esto en Python.
    frecuencias = Counter(palabras)

    # Devolver el objeto Counter
    return frecuencias

# --- Bloque Principal de Ejecuci칩n ---

# # Define el nombre del archivo de entrada.
nombre_del_archivo = "mi_documento.txt"

# # Llama a la funci칩n principal.
conteo = contador_de_palabras(nombre_del_archivo)

# # Si el conteo fue exitoso (no es None), imprime los resultados.
if conteo:
    # # El m칠todo `most_common()` devuelve una lista de tuplas (palabra, frecuencia)
    # # # ordenada por frecuencia, de mayor a menor.
    print(f"\n# Resultados del an치lisis de frecuencia de palabras en '{nombre_del_archivo}':")
    print("# ----------------------------------------------------")
    
    # # Itera sobre las palabras y frecuencias e imprime el resultado
    # # # Se usa un slice `[:10]` para mostrar solo las 10 palabras m치s comunes.
    palabras_mas_comunes = conteo.most_common(10)
    
    # # Si hay palabras para mostrar:
    if palabras_mas_comunes:
        # # Imprime un encabezado para las palabras m치s comunes
        print("# Las 10 palabras m치s comunes son:")
        for palabra, frecuencia in palabras_mas_comunes:
            # # Imprime cada palabra y su conteo
            print(f"# -> '{palabra}': {frecuencia} veces")

        # # Tambi칠n se puede imprimir el conteo total de palabras 칰nicas:
        print(f"\n# El total de palabras 칰nicas encontradas es: {len(conteo)}")
    else:
        # # Mensaje si el archivo estaba vac칤o despu칠s de la limpieza
        print("# El archivo estaba vac칤o o no conten칤a palabras despu칠s del preprocesamiento.")

```
