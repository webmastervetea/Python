```python
import calendar
import os
import sys
import tkinter as tk
from tkinter import messagebox, simpledialog

# --- FUNCI√ìN PRINCIPAL DE GENERACI√ìN ---

def generar_calendario_mensual(mes, anio):
    """
    Genera y devuelve el calendario de texto para el mes y a√±o dados.

    :param mes: N√∫mero del mes (1-12).
    :param anio: N√∫mero del a√±o (Ej: 2025).
    :return: Una cadena de texto con el calendario.
    """
    try:
        # # Crear una instancia de TextCalendar
        cal = calendar.TextCalendar(calendar.MONDAY) # Iniciar la semana en Lunes (Opci√≥n: calendar.SUNDAY)
        
        # # Formatear el mes. El resultado es una cadena de texto lista para imprimir.
        calendario_texto = cal.formatmonth(anio, mes, w=2, l=1)
        
        return calendario_texto
        
    except ValueError:
        return "‚ùå ERROR: Mes o a√±o fuera de rango. Aseg√∫rate de que el mes est√© entre 1 y 12."
    except Exception as e:
        return f"‚ùå ERROR inesperado: {e}"

def solicitar_datos_y_mostrar_calendario():
    """
    Pide al usuario el mes y el a√±o y muestra el calendario.
    """
    
    # 1. Pedir el a√±o
    try:
        anio_entrada = simpledialog.askinteger("Generador de Calendario", "Introduce el a√±o (Ej: 2025):", minvalue=1900, maxvalue=2100)
        if anio_entrada is None: 
            print("# Operaci√≥n cancelada por el usuario.")
            return

        # 2. Pedir el mes
        mes_entrada = simpledialog.askinteger("Generador de Calendario", "Introduce el n√∫mero del mes (1-12):", minvalue=1, maxvalue=12)
        if mes_entrada is None: 
            print("# Operaci√≥n cancelada por el usuario.")
            return

    except Exception:
        messagebox.showerror("Error de Entrada", "Aseg√∫rate de introducir n√∫meros enteros v√°lidos.")
        return
        
    # 3. Generar el calendario
    calendario = generar_calendario_mensual(mes_entrada, anio_entrada)
    
    # 4. Mostrar el resultado en la consola
    print(f"\n# --- CALENDARIO DE {calendar.month_name[mes_entrada].upper()} de {anio_entrada} ---")
    print(calendario)
    print("----------------------------------------------------------")
    
    # 5. Opcionalmente, mostrar el resultado tambi√©n en un cuadro de di√°logo
    messagebox.showinfo(f"Calendario de {calendar.month_name[mes_entrada]}", calendario)


# --- Bloque Principal de Ejecuci√≥n ---

def main():
    # # Inicializar Tkinter para los di√°logos gr√°ficos
    root = tk.Tk()
    root.withdraw() 
    
    print("# --- GENERADOR DE CALENDARIO MENSUAL ---")
    
    solicitar_datos_y_mostrar_calendario()

if __name__ == "__main__":
    main()
```
