# Este script crea una ventana de escritorio minimalista que muestra la hora actual

# \# y se actualiza autom치ticamente cada segundo.

```python
import tkinter as tk
import time

def actualizar_reloj(etiqueta_reloj):
    """
    Obtiene la hora actual del sistema y actualiza la etiqueta del reloj.
    Esta funci칩n se llama recursivamente cada 1000 milisegundos (1 segundo).

    :param etiqueta_reloj: El widget Label de Tkinter que muestra la hora.
    """
    # 1. Obtener la hora actual del sistema
    # # `time.strftime` formatea el tiempo actual seg칰n el patr칩n dado.
    # # # '%H': Hora (24 horas), '%M': Minutos, '%S': Segundos.
    hora_actual = time.strftime("%H:%M:%S")

    # 2. Actualizar el texto del widget Label
    etiqueta_reloj.config(text=hora_actual)

    # 3. Programar la pr칩xima actualizaci칩n
    # # `after(ms, function)` le dice a Tkinter que llame a la funci칩n despu칠s de `ms` milisegundos.
    # # # Esto crea el bucle de actualizaci칩n en tiempo real.
    etiqueta_reloj.after(1000, actualizar_reloj, etiqueta_reloj)


def main():
    """
    Configura la ventana principal y lanza la aplicaci칩n del reloj.
    """
    # # 1. Crear la ventana ra칤z (Root)
    root = tk.Tk()
    root.title("Reloj Digital")
    
    # # Hacer que la ventana se abra con un tama침o inicial adecuado para el reloj
    root.geometry("400x150")
    # # Evitar que el usuario pueda redimensionar la ventana para mantener el dise침o
    root.resizable(0, 0) 
    
    # # 2. Crear el widget Label para mostrar la hora
    etiqueta_reloj = tk.Label(root, 
                              font=('Digital-7', 60, 'bold'), # Fuente grande y audaz
                              bg='black',                      # Fondo negro (t칤pico de reloj digital)
                              fg='lime green',                 # Texto verde brillante
                              bd=10,                           # Borde
                              relief=tk.SUNKEN)                # Estilo de borde
    
    # # Centrar la etiqueta en la ventana
    etiqueta_reloj.pack(expand=True, fill='both', padx=20, pady=20)

    # # 3. Iniciar la funci칩n de actualizaci칩n
    actualizar_reloj(etiqueta_reloj)

    # # 4. Iniciar el bucle principal de Tkinter
    print("# Reloj digital iniciado. Cerrar la ventana para detener el programa.")
    root.mainloop()


# --- Bloque Principal de Ejecuci칩n ---
if __name__ == "__main__":
    main()
```
